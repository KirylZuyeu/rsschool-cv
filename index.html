                    void checkHasNegavite(double* arr, int size, int *posNegFirst, int *posNegSecond) {
                      bool checkNegativFirst = false;
                      bool checkNegativSecond = false;

                      int indexSecond;

                        if (*(arr + i) < 0) {

                            checkNegativFirst = true;
                            indexFirst = i;
                            continue;
                          }
                          else {
                            checkNegativSecond = true;

                            break;
                          }
                        }
